import os
from dotenv import load_dotenv

load_dotenv()

BOT_TOKEN = os.getenv("BOT_TOKEN", "")
API_BASE_URL = os.getenv("API_BASE_URL", "")
API_KEY = os.getenv("API_KEY", "")

if not BOT_TOKEN:
    raise ValueError("BOT_TOKEN missing")
if not API_BASE_URL:
    raise ValueError("API_BASE_URL missing")
